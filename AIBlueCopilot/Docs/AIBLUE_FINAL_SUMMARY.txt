## ?? COMMAND QUICK REFERENCE

```cpp
// Get copilot (creates if needed)
AAIBlueCopilot* C = UAIBlueCopilotLibrary::GetCopilotInstance(W);

// Simple commands
UAIBlueCopilotLibrary::BuildScene(W, TEXT("Description"));
UAIBlueCopilotLibrary::BuildUI(W, TEXT("Description"));
UAIBlueCopilotLibrary::AnalyzeArchitecture(W);
UAIBlueCopilotLibrary::RefactorBlueprint(W, TEXT("/Path"), TEXT("Request"));

// Check status
if (C->IsCopilotReady()) { /* ready */ }

// Direct command
FCopilotCommand Cmd;
Cmd.CommandType = ECopilotCommandType::BuildScene;
Cmd.CommandText = TEXT("Your prompt");
C->ExecuteCommand(Cmd);

// Handle response
C->OnCopilotResponse.AddDynamic(this, &AMyActor::OnResponse);

// Get last response
FString Response = C->GetCopilotResponse();
```
